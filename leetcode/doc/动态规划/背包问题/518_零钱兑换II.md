## 提交代码

我可以理解01背包的滚动数组，但是我理解不了完全背包的滚动数组。

理解不了，便用二维数组。虽然二维数组多用了些空间，但是代码很直观。

01背包，对一个物品的选择是，不放 or 放。

完全背包，对一个物品的选择是，不放 or 放一个 or 放两个.... 仅此而已。所以在掌握01背包之后，完全背包很容易实现。

本题略微有点难度的地方是，当背包容量为0时，需要初始化为1，表示仅使用某种硬币，可以完全找零。(自行调试，体会)

```c++
#include <vector>
#include <iostream>

using namespace std;

class Solution {
public:
    int change(int amount, vector<int>& coins) {
        // 思路：完全背包问题
        // 无法放下；可以放下：不放，放1一个，两个，....

        int kinds = coins.size();

        // dp[i][j]表示，前i种硬币，凑凑恰好为j，的组合数
        // 整体初始化为零，默认全部找不开
        // 初始化第一列，当需要找0元。表示使用某一种硬币可以全部找开。
        // 初始化第一行，使用第一种硬币，凑amount；如果可以凑上，初始化为1
        vector<vector<int>> dp(kinds,vector<int>(amount+1,0));
        for(int i=0; i<kinds; i++)
            dp[i][0] = 1;
        for(int i=1; i<=amount; i++){
            if(i%coins[0]==0)
                dp[0][i] = 1;
        }

        for(int i=1; i<kinds; i++){ // 遍历物品
            for(int j=1; j<=amount; j++){ // 遍历背包
                if(coins[i] > j)
                    dp[i][j] = dp[i-1][j]; // 当前物品无法放入，相当于使用前i-1种物品，恰好放入j容量
                else{
                    // 可以放入:选择不放，放入一个，放入两个....
                    int cnt = j/coins[i];
                    int max_method = dp[i-1][j];
                    for(int k=1; k<=cnt; k++){
                        max_method += dp[i-1][j-k*coins[i]];
                    }
                    dp[i][j] = max_method;
                }
                
            }
        }

        // for(auto vec:dp){
        //     for(auto v : vec)
        //         cout<<v<<" ";
        //     cout<<endl;
        // }

        return dp[kinds-1][amount];
    }
};
```
